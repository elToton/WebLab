                void RemoveDups(char* pStr)
                {
                    const char* src = pStr;
                    while (*src)
                    {
                        if (pStr != src)
                            *pStr = *src;
                        ++pStr;
                        while ((*src + 1) && *(src + 1) == *src)
                        {
                            ++src;
                        }
                        ++src;
                    }

                    *pStr = '\0';
                }
